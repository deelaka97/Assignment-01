const readline = require('readline');

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

let assignmentName;
let studentsMarks = [];

// Function to input assignment name
function inputAssignmentName() {
  rl.question('Enter the assignment name: ', (name) => {
    assignmentName = name;

    // Proceed to input students' marks
    inputStudentMarks(0);
  });
}

// Function to input students' marks
function inputStudentMarks(index) {
  if (index >= 30) {
    rl.close();
    processMarks();
    return;
  }

  rl.question(`Enter the mark for student ${index + 1}: `, (mark) => {
    mark = parseInt(mark);
    if (isValidMark(mark)) {
      studentsMarks.push(mark);
      inputStudentMarks(index + 1);
    } else {
      console.log('Invalid input. Please enter a mark between 0 and 30.');
      inputStudentMarks(index);
    }
  });
}

// Function to check if a mark is valid
function isValidMark(mark) {
  return !isNaN(mark) && mark >= 0 && mark <= 30;
}

// Function to process and print assignment name and student marks
function processMarks() {
  console.log('Assignment Name:', assignmentName);
  console.log("Students' Marks:", studentsMarks);
}

// Start by inputting the assignment name
inputAssignmentName();
