// Function to calculate the mean and standard deviation
function calculateMeanAndStandardDeviation(studentsMarks) {
  let sum = 0;
  let squaredDifferencesSum = 0;

  // Calculate the sum of marks
  for (const mark of studentsMarks) {
    sum += mark;
  }

  // Calculate the mean (average)
  const mean = sum / studentsMarks.length;

  // Calculate the sum of squared differences
  for (const mark of studentsMarks) {
    const diff = mark - mean;
    squaredDifferencesSum += diff * diff;
  }

  // Calculate the variance and standard deviation
  const variance = squaredDifferencesSum / studentsMarks.length;
  const standardDeviation = Math.sqrt(variance);

  // Return the results
  return { mean, standardDeviation };
}

// Print results of studentsMarks array
const { mean, standardDeviation } = calculateMeanAndStandardDeviation(studentsMarks);
console.log('Mean:', mean);
console.log('Standard Deviation:', standardDeviation);
